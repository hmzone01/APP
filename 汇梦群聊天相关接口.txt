### 1.用户申请入群
####用户在App端申请加入项目群，系统自动允许用户进入所申请的群,入群后刷新个人群列表。
url: http://website/chat/group!joinGroup
method: post
Request:{
	headers{
		uid:"1000221",#用户ID
    		accessToken:"",#用户token
    		client:"F9FQXGP8FCM5",#客户端标识
    		expire:1411527886457#失效时间
	}
	data{
	       “groupId”:”22321”# 群ID
	}
}
Response:{
	headers{
    		uid:"1000221",#用户ID
    		accessToken:"",#用户token
    		client:"F9FQXGP8FCM5",#客户端标识
    		expire:1411527886457#失效时间
	}
	data{
  		"success": true, #是否成功，true为成功，false为失败。如果为false,“info”表示失败原因。
  		"info": “申请入群成功”,#提示信息
  	}
}

### 4.用户用户设置在群中显示的名称
####当前登陆用户根据自己的ID和群ID，设置个人在群中的名称。只能修改自己的显示名称
url: http://website/chat/group!userSetGroup
method: post
Request:{
	headers{
		uid:"1000221",#用户ID
    		accessToken:"",#用户token
    		client:"F9FQXGP8FCM5",#客户端标识
    		expire:1411527886457#失效时间
	}
	data{
	       “groupId”:”22321”# 群ID
		“showName”:”彪彪小王子”#在群中显示的名称,可以为空，为空表示不设置，管理员可以为别人设置，自己可以给自己设置
	}
}
Response:{
	headers{
    		uid:"1000221",#用户ID
    		accessToken:"",#用户token
    		client:"F9FQXGP8FCM5",#客户端标识
    		expire:1411527886457#失效时间
	}
	data{
  		"success": true, #是否成功，true为成功，false为失败。如果为false,“info”表示失败原因。
  		"info": “设置群显示名称成功”,#提示信息
  	}
}

### 6.踢出群成员

url: http://website/chat/group!deleteUser
method: post
Request:{
	headers{
		uid:"1000221",#用户ID
    		accessToken:"",#用户token
    		client:"F9FQXGP8FCM5",#客户端标识
    		expire:1411527886457#失效时间
	}
	data{
	       “groupId”:”22321”# 群ID
	       “groupUserId”:”12555”,#要删除群成员ID
	}
}
Response:{
	headers{
    		uid:"1000221",#用户ID
    		accessToken:"",#用户token
    		client:"F9FQXGP8FCM5",#客户端标识
    		expire:1411527886457#失效时间
	}
	data{
  		"success": true, #是否成功，true为成功，false为失败。如果为false,“info”表示失败原因。
  		"info": “已成功踢出成员XX”,#提示信息,XX表示成员名称
  	}

### 7.设置群资料
#### 根据群ID和用户ID修改群公告、群介绍，群名称。只有管理员可以使用此功能，非管理员不能使用。
url: http://website/chat/group!updateGroup
method: post
Request:{
	headers{
		uid:"1000221",#用户ID
    		accessToken:"",#用户token
    		client:"F9FQXGP8FCM5",#客户端标识
    		expire:1411527886457#失效时间
	}
	data{
	       “groupId”:”22321”# 群ID
	       “groupName”:”悦尚九鲜股东群”,#新的群名称，如果为空则表示不修改
 	       “type”:”群类别”,#群类别，为空表示不修改
               “introduction”:”这是悦尚九鲜股东群”,#群介绍，为空表示不修改
               “notice”:“这是群公告内容”#群公告,为空表示不修改	
	}
}
Response:{
	headers{
    		uid:"1000221",#用户ID
    		accessToken:"",#用户token
    		client:"F9FQXGP8FCM5",#客户端标识
    		expire:1411527886457#失效时间
	}
	data{
  		"success": true, #是否成功，true为成功，false为失败。如果为false,“info”表示失败原因。
  		"info": “修改群资料成功”,#提示信息
  	}
}
